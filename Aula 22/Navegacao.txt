@Composable
fun GerenciadorTelas() {

    // NavControler (Controle das rotas/telas e histórico de acesso as telas)
    val navController = rememberNavController()

    // Coluna
    Column() {
        // Exibir o histórico de rotas
        Box() {

            // Obter o nome de todas as telas (rotas) acessadas
            val telas by navController.currentBackStack.collectAsState()

            // Mapear dados
            val nomesTela = telas.mapNotNull { it.destination.route }

            // Exibir
            Text(
                nomesTela.toString(),
                modifier = Modifier.padding(30.dp)
            )
        }

        // Exibir as telas
        NavHost(navController = navController, startDestination = "tela1") {

            // Tela 1
            composable("tela1") {
                Tela1(
                    exibirTela2 = { navController.navigate("tela2") },
                    exibirTela3 = { navController.navigate("tela3/Enviado via Tela 1") }
                )
            }

            // Tela 2
            composable("tela2") {
                Tela2(
                    exibirTela1 = { navController.navigate("tela1") },
                    exibirTela3 = { navController.navigate("tela3/Enviado via Tela 2") }
                )
            }

            // Tela 3
            composable("tela3/{parametro}") {

                // Extrair o valor do parâmetro
                backStackEntry -> val parametro = backStackEntry.arguments?.getString("parametro")

                Tela3(
                    exibirTela1 = { navController.navigate("tela1") },
                    exibirTela2 = { navController.navigate("tela2") },
                    voltar = { navController.popBackStack() },
                    exibirTela1RemoverHistorico = { navController.navigate("tela1") { popUpTo(0) } },
                    texto = parametro.toString()
                )
            }



        }
    }

}








@Composable
fun Tela1(
    exibirTela2: () -> Unit,
    exibirTela3: () -> Unit
) {
    Column(
        modifier = Modifier.fillMaxSize(),
        verticalArrangement = Arrangement.Center,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Text(
            text = "Tela 1",
            fontSize = 25.sp
        )

        Spacer(modifier = Modifier.height(20.dp))

        Button(
            onClick = exibirTela2
        ) { Text("Tela 2") }

        Button(
            onClick = exibirTela3
        ) { Text("Tela 3") }

    }
}

@Composable
fun Tela3(
    exibirTela1: () -> Unit,
    exibirTela2: () -> Unit,
    voltar: () -> Unit,
    exibirTela1RemoverHistorico: () -> Unit,
    texto: String
) {
    Column(
        modifier = Modifier.fillMaxSize(),
        verticalArrangement = Arrangement.Center,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Text(
            text = "Tela 3",
            fontSize = 25.sp
        )

        Text(
            text = "Parâmetro recebido: $texto",
            fontSize = 20.sp,
            color = Color.Magenta
        )

        Spacer(modifier = Modifier.height(20.dp))

        Button(
            onClick = exibirTela1
        ) { Text("Tela 1") }

        Button(
            onClick = exibirTela2
        ) { Text("Tela 2") }

        Button(
            onClick = voltar
        ) { Text("Voltar") }

        Button(
            onClick = exibirTela1RemoverHistorico
        ) { Text("Tela 1 - Remover histórico") }
    }
}
