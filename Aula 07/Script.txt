// Função utilizando o Smart Cast (is)
fun somaValor (numero : Any) {
    if (numero is Int) {
        println(numero + 10)
    } else if (numero is Double) {
        println(numero + 15)
    } else {
        println("Apenas são aceitos números do tipo Int ou Double")
    }
}
